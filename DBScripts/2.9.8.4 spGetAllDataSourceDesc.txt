CREATE PROC spGetAllDataSourceDesc
	@ENV_ID int
AS
BEGIN
SELECT 'DATABASE' AS TYPE, DB_ID,[DB_ENV_ID],[DB_NAME],[VEN_IMAGE_SRC] FROM [DATABASE] INNER JOIN [VENDER] ON [DB_VENDER_ID] = [VEN_ID]
WHERE DB_ENV_ID = @ENV_ID
UNION
SELECT 'MIDDLEWARE' AS TYPE, MDW_ID,[MDW_ENV_ID],[MDW_NAME],[VEN_IMAGE_SRC] FROM [MIDDLEWARE] INNER JOIN [VENDER] ON [MDW_VENDER_ID] = [VEN_ID]
WHERE MDW_ENV_ID = @ENV_ID
END
